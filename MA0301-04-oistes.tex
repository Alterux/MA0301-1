\begin{problem}[7]
  Simplify $\bigl( p \wedge (\neg s \vee q \vee \neg q) \bigr) \vee \bigl( (s
  \vee t \vee \neg s) \wedge \neg q \bigr)$.
\end{problem}

\begin{answer}
  This is the same as the question about the circuit diagram in the previous
  exercise. Since $(\neg s \vee q \vee \neg q) = (\neg s \vee T_0) = T_0$
  and similarly, $(s \vee t \vee \neg s) = (t \vee T_0) = T_0$ we have
  %
  \begin{equation*}
    \bigl( p \wedge (\neg s \vee q \vee \neg q) \bigr) \vee \bigl( (s
    \vee t \vee \neg s) \wedge \neg q \bigr)
    = \bigl( p \wedge T_0 \bigr) \vee \bigl( T_0 \wedge \neg q \bigr)
    = p \vee \neg q
  \end{equation*}
\end{answer}

\end{subproblem} 

\begin{subproblem}
  $xy + w$
\end{subproblem} 

\begin{answer}
  The expression above is dependent on the value of $y$ as $x \cdot \1 = 1$, but
  $x \cdot \0 = \0$. Thus, the expression is true when $y \vee w$ is true,

\end{answer}

\begin{problem}
  Find the values of the Boolean variables $w$, $x$, $y$, $z$ that satisfies the
  following system of simultaneous equations.
  %
  \begin{align}
    x + \overline{x} y & = 0 \label{eq:12-a} \\
    \overline{x} y & = \overline{x} z \label{eq:12-b} \\ 
    \overline{x} y + \overline{x} \overline{z} + z w & = \overline{z} w \label{eq:12-c} 
  \end{align}
\end{problem}

\begin{answer}
  For \cref{eq:12-a} to be false both $x$ and $\overline{x}y$ must be false,
  the first implying $x$ is false, and the second that $y$ is false (as
  $\overline{x}$ is true).

  Inserting this into \cref{eq:12-b} we get
  %
  \begin{equation*}
    \overline{\0} \cdot \0 = \overline{1} z
  \end{equation*}
  %
  and solving this equation for $z$ we get $z = \0$. Inserting this into the
  last \cref{eq:12-c}
  %
  \begin{equation*}
    \overline{\0} \cdot \0 + \overline{x} \cdot \overline{\0} + \0 \cdot w = \overline{z} w\,.
  \end{equation*}
  %
  As the right left-side is $0$ we get $w = \1$. In summary $(x,y,z,w)=(\0,\0,\0,\1)$.
\end{answer}

\begin{problem}[17]
  Using the laws of set theory, simplify each of the following:
\end{problem}

\begin{subproblem}[2]
  \ifthenelse{\boolean{\solutions}}{}{%
  $(A \cap B) \cup (A \cap B \cap \overline{C} \cap D) \cup (\overline{A} \cap B)$}
\end{subproblem}

\begin{answer} 
  \begin{align*}
    \\[-2.5cm]
      & \textbf{Steps} && \textbf{Reasons} \\
      & (A \cap B) \cup (A \cap B \cap \overline{C} \cap D) \cup (\overline{A} \cap B)
     && \\
    = & (A \cap B) \cup (\overline{A} \cap B)  
     && \text{Absorption law} \\
    = &(A \cap \overline{A}) \cup B
     && \text{Distributive law} \\
    = & \emptyset \cup B
     && \text{Inverse laws} \\
    = & B 
     && \text{Identity laws}
  \end{align*}
\end{answer}

\begin{subproblem}[4]
  % \ifthenelse{\equal{solutions}{true}}{TRUE}{}
  \ifthenelse{\boolean{\solutions}}{}{%
    $\overline{A} \cup \overline{B} \cup (A \cap B \cap \overline{C})$}
\end{subproblem}

\begin{answer} 
  \begin{align*}
    \\[-2.5cm]
      & \textbf{Steps} && \textbf{Reasons} \\
      & \overline{A} \cup \overline{B} \cup (A \cap B \cap \overline{C})
     && \\
    = & \bigl[(\overline{A} \cup \overline{B}) \cup (A \cap B)\bigr] \cap \bigl[\overline{A} \cup \overline{B} \cup \overline{C}\bigr]
     && \text{Distributive laws twice} \\
    = & \bigl[ \overline{A \cap B} \cup (A \cap B)\bigr] \cap \bigl[\overline{A} \cup \overline{B} \cup \overline{C}\bigr]
     && \text{DeMorgan's laws} \\ 
    = & \bigl[ \U \bigr] \cap \bigl[\overline{A} \cup \overline{B} \cup \overline{C})\bigr]
     && \text{Inverse laws} \\
    = & \overline{A} \cup \overline{B} \cup \overline{C}
     && \text{Domination laws}
  \end{align*}
\end{answer}
